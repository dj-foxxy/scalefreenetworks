\headerbox{Problem}{name=problem,column=0,row=0}
{
    The degrees of nodes in Scale-free networks follow a power-law distribution.
}
\headerbox{Exact algorithm}{name=exact-algorithm,
below=problem,column=0,span=1}
{
    \begin{algorithm}[H]
        \KwIn{A graph $G = (V, E)$, with $N$ nodes}
        \KwOut{The clustering coefficient $C$, for $G$}
        $c \gets 0$\\
        \For{$i \in V$}
        {
            $d \gets i.degree$\\
            $e \gets 0$\\
            \For{$j \in i.neighbours$}
            {
                \For{$k \in j.neighbours$}
                {
                    \If {$i\mathtt{isneighbour}\left(i,j\right)$}
                    {
                        $e \gets e + 1$
                    }
                }
            }
            $c \gets \frac{e}{d\left(d-1\right)}$
        }
        \KwRet{$\frac{c}{N}$}
        \caption{NODECOUNT}
    \end{algorithm}
}
\headerbox{Approximation Algorithm}{name=approximation-algorithm,
below=problem, column=1, row=1}
{
    \begin{algorithm}[H]
        \KwIn{Integer $k$ and a graph $G = \left(V,E\right)$ with $N$ nodes}
        \KwOut{An approximation of the clustering coefficient $C$ of $G$}
        $c \gets 0$\\
        \For{$i \in k$}
        {
        $j \gets\mathtt{UniformRandomNode}\left(\right)$\\
        $u \gets\mathtt{UnifromRandomNeighbour}\left(j\right)$\\
        \Repeat{$u \neq v$}
        {
            $v \gets \mathtt{UniformRandomNeighbour}\left(u\right)$\\
        }
        \If{$\mathtt{EdgeExists}\left(i, v\right)$}
        {
            $c \gets c + 1$\\
        }
    }
        \KwRet{$\frac{c}{k}$}
        \caption{Schank-Wagner}
    \end{algorithm}
}
